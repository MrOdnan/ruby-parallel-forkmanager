    def finish(exit_code = 0)
        if self.has_block
            raise "Do not use finish(...) when using blocks.  Use an explicit exit in your block instead!\n"
        end

        if self.in_child
            exit_code ||= 0
            Kernel.exit!(exit_code)
        end

        if self.max_proc == 0
            on_finish($$, exit_code, self.processes[$$], 0, 0)
            self.processes.delete($$)
        end
        return 0
    end

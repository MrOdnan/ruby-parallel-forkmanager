    def on_start(*params)
        begin
            if @do_on_start.class().name == 'Proc'
                my_argc = @do_on_start.arity - 1
                if my_argc > 0
                    my_params = params[0 .. my_argc]    
                else
                    my_params = params[0]
                end
                params = my_params
                @do_on_start.call(*params)
            end
        rescue
            raise "on_start failed\n"
        end       
    end

    def _NT_waitpid(pid, par)
        if par == Process::WNOHANG
            pids = @processes.keys()
            if pids.length() == 0
                return -1
            end
            
            kid = 0
            for my_pid in pids
                kid = waitpid(my_pid, par)
                if kid != 0
                    return kid
                end
            return kid
            end
        else
            return waitpid(pid, par)    
        end
    end

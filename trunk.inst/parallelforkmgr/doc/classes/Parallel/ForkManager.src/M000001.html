    def initialize(procs)
        @debug = 0
        @max_proc = procs
        @processes = {}
        @do_on_finish = {}
        @in_child = false
        @has_block = false
        @on_wait_period = nil

        if self.debug == 1
            print "in initialize #{max_proc}!\n"
        end
    end
